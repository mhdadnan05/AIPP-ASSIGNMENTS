def sum_of_even_and_odd_numbers(numbers_list):
  """
  Calculates the sum of even and odd numbers within a given list.

  This function iterates through a list of numbers, categorizing each
  number as either even or odd and accumulating their respective sums.
  It also includes type checking to ensure valid input.

  Args:
    numbers_list: A list containing integer numbers.

  Returns:
    A tuple containing two integers: (even_nums_sum, odd_nums_sum) if the
    input is valid and processing is successful.
    Returns False if the input `numbers_list` is not a list, or if any
    element within the list is not an integer.
  """
  # Check if the input is a list
  if type(numbers_list) == type(list()):
    # Initialize variables to store the sum of even and odd numbers
    even_nums_sum = 0
    odd_nums_sum = 0
    # Iterate through each element in the provided list
    for number in numbers_list:
      # Check if the current element is an integer
      if type(number) == type(int()):
        # Determine if the number is even or odd
        if number % 2 == 0:
          # Add to even sum if it's an even number
          even_nums_sum += number
        else:
          # Add to odd sum if it's an odd number
          odd_nums_sum += number
      else:
        # Return False if any element in the list is not an integer
        return False
    # Return the accumulated sums of even and odd numbers
    return even_nums_sum, odd_nums_sum
  else:
    # Return False if the input is not a list
    return False
```

class Sample_student:
  """
  This is a class that manages an individual student's data.

  Attributes:
    name (str): The name of the student.
    roll_no (int): The roll number of the student.
    hostel_status (str): The hostel status of the student (e.g., 'resident', 'day scholar').
    fee_paid (int): The total fee paid by the student.
  """
  def __init__(self,name,roll_no,hostel_status):
    """
    Initializes a new Sample_student instance.

    Args:
      name (str): The name of the student.
      roll_no (int): The roll number of the student.
      hostel_status (str): The hostel status of the student.

    Returns: None
    """
    # Assign the provided name to the student object
    self.name = name
    # Assign the provided roll number to the student object
    self.roll_no = roll_no
    # Assign the provided hostel status to the student object
    self.hostel_status = hostel_status
    # Initialize the fee paid to 0 for a new student
    self.fee_paid = 0

  def fee_update(self, fee):
    """
    Updates the fee paid by the student.

    Args:
      fee (int): The amount of fee paid by the student. Must be a positive integer.

    Returns:
      bool: True if the fee update was successful, False otherwise.
    """
    # Check if the provided fee is an integer and is positive
    if type(fee) == type(int()) and fee > 0:
      # Add the paid fee to the total fee_paid
      self.fee_paid += fee
      # Indicate successful update
      return True
    else:
      # Print an error message for invalid input
      print("Invalid input")
      # Indicate unsuccessful update
      return False

  def display_details(self):
    """
    Displays the details of the student.

    This includes the student's name, roll number, hostel status, and total fee paid.

    Args: None

    Returns: None
    """
    # Print the student's details using an f-string for formatted output
